329 void reverse_transform_data(FILE *input_file, const char *output_file, const char *output_status) {

337  //Variable for holding reading while reading.

339  //Variable for reading long long int from input.

343 
344  //Opening the file for appending.
345  file = fopen(output_file, "wb");
346  if(file == NULL) {
347  printf("Error in opening the file for writing.\n");
348  return;
349  }
350 
351  in_data.letter_output_map = NULL;
352  in_data.index_output_map = NULL;
353  in_data.letter_jump_map = NULL;
354  in_data.index_jump_map = NULL;
355  fseek(input_file, 0, SEEK_SET);
356  //Reading the data.
357  c = fgetc(input_file);
358  fseek(input_file, -1L, SEEK_CUR);
359  while (c > 0) {
360  if((status == INDEX_OUTPUT || status == INDEX_JUMP)) {
361  indices = 0LL;
362  c = fgetc(input_file);
363  while(c != 32 && c != 10) {
364  indices = indices*10LL + (long long int)(c - 48);
365  c = fgetc(input_file);
366  }
367  } else {
368  c = fgetc(input_file);
369  if(c == 32) {
370  c = fgetc(input_file);
371  }
372  }
373  if(c != 10 && c > 0) {
374  if(status == LETTER_OUTPUT) {
375  in_data.letter_output_map = append_letter(c, in_data.letter_output_map);
376  } else if(status == LETTER_JUMP) {
377  in_data.letter_jump_map = append_letter(c, in_data.letter_jump_map);
378  } else if(status == INDEX_OUTPUT) {
379  in_data.index_output_map = append_index(indices, in_data.index_output_map);
380  } else {
381  in_data.index_jump_map = append_index(indices, in_data.index_jump_map);
382  }
383  } else if(c == 10){
384  if(status == LETTER_OUTPUT) {
385  status = INDEX_OUTPUT;
386  } else if(status == INDEX_OUTPUT) {
387  status = LETTER_JUMP;
388  } else if(status == LETTER_JUMP) {
389  status = INDEX_JUMP;
390  } else {
391  status = LETTER_OUTPUT;
392  output = elloborate(in_data.letter_output_map, in_data.index_output_map);
393  if(in_data.letter_jump_map != NULL) {
394  output = reverse_jumps(in_data.letter_jump_map, in_data.index_jump_map, output);
395  }
396  print_map_letter_to_file_as_plaintext(output, file);
397  in_size += map_length_letter(output);
398  theoretical_out_size += map_length_letter(in_data.letter_output_map);
399  theoretical_out_size += map_length_letter(in_data.letter_jump_map);
400  theoretical_out_size += map_length_index(in_data.index_output_map);
401  theoretical_out_size += map_length_index(in_data.index_jump_map);
402  original_out_size += map_length_letter(in_data.letter_output_map)*2+1LL;
403  original_out_size += map_length_letter(in_data.letter_jump_map)*2+3LL;
404  original_out_size += map_length_index(in_data.index_output_map)*2+1LL;
405  original_out_size += map_length_index(in_data.index_jump_map)*2+1LL;
406  re_initialize_transform_variables();
407  delete_map_letter(output);
408  output = NULL;
409  }
410  }
411  }
412  fclose(file);
413  file = fopen(output_status, "wb");
414  if(file == NULL) {
415  printf("Error in opening the file for updating the status.\n");
416  return;
417  }
418  original_out_size += 1LL;
419  fprintf(file, "Decompression was successful.\n");
420  fprintf(file, "Input Size : %lld bytes\n", in_size);
421  fprintf(file, "Theoretical Output Size : %lld bytes\n", theoretical_out_size);
422  fprintf(file, "Original Output Size : %lld bytes\n", original_out_size);
423  fprintf(file, "Theoretical Efficiency : %lld %%\n", (100LL - (theoretical_out_size*100LL/in_size)));
424  fprintf(file, "Original Efficiency : %lld %%\n", (100LL - (original_out_size*100LL/in_size)));
425  fclose(file);
426 }
