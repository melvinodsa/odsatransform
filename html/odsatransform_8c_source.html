364  c = fgetc(input_file);

369  } else if(status == INDEX_JUMP){
370  in_data.index_jump_map = append_index(indices, in_data.index_jump_map);
371  }
372  } else if(c > 0 && (status == LETTER_OUTPUT || status == LETTER_JUMP)) {
373  if(status == LETTER_OUTPUT) {
374  in_data.letter_output_map = append_letter(c, in_data.letter_output_map);
375  } else if(status == LETTER_JUMP) {
376  in_data.letter_jump_map = append_letter(c, in_data.letter_jump_map);
377  }
378  } else if(c == -2){
379  if(status == LETTER_OUTPUT) {
380  status = INDEX_OUTPUT;
381  } else if(status == LETTER_JUMP) {
382  status = INDEX_JUMP;
383  }
384  } else if(c == 10){
385  if(status == INDEX_OUTPUT) {
386  status = LETTER_JUMP;
387  } else if(status == INDEX_JUMP) {

409  c = 10;
